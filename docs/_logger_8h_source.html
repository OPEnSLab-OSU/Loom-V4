223 void startFunction(const char* file, const char* func, unsigned long num, int freeMemory){
224 if(enableFunctionSummaries){
225 SLOG(func);
226 // Log the start time of the function
227 char fileName[260];
228 memset(fileName, '\0', 260);
229 truncateFileName(file, fileName);
230 struct functionInfo* newFunction = (struct functionInfo*) malloc(sizeof(struct functionInfo));
231 strncpy(newFunction->fileName, fileName, 260);
232 strncpy(newFunction->funcName, func, 260);
233 newFunction->lineNumber = num;
234 newFunction->netMemoryUsage = freeMemory;
235 newFunction->time = millis();
236 newFunction->indentCount = callStack.size();

245 void endFunction(int freeMemory){
246 if(enableFunctionSummaries){
247 char fileName[100];
248 char file[260];
249 char func[260];
250 char output[300];
251 char indents[10];
252
253 // 0 fill the indents array
254 memset(indents, '\0', 10);
255
256 struct functionInfo* info = callStack.front();
257 int memUsage = info->netMemoryUsage;
258 unsigned long time = millis() - info->time;
259 int percentage = ((float)memUsage / 32000.0) * 100;
260
261 // Copy to internal variables
262 strncpy(file, info->fileName, 260);
263 strncpy(func, info->funcName, 260);
264
265 // Cap the number of inents at the size of the array
266 if(info->indentCount > 10){
267 info->indentCount = 10;
268 }
269
270 // Set the number of tab characters we need to add
271 memset(indents, '\t', info->indentCount);
272
273 // Pop the last function off the call stack
274 free(info);
275 callStack.pop();
276
277 // Format the fileName and log output, this function uses 976 bytes
278 snprintf_P(fileName, 100,PSTR("/debug/funcSummaries_%i.log"), sdInst->getCurrentFileNumber());
279 snprintf_P(output, 300, PSTR("%s[%s:%s] Summary\n%s\tInitial Free Memory: %i B (%i %% Free)\n%s\tEnding Free Memory: %i B\n%s\tNet Usage: %i B\n%s\tElapsed Time: %u MS"), indents, file, func, indents, memUsage, percentage, indents, freeMemory, indents, memUsage-freeMemory, indents, time);
280
281 // Log as long as we have given it a SD card instance
282 if(sdInst != nullptr && enableSDLogging)
283 sdInst->writeLineToFile(fileName, output);
284 }
